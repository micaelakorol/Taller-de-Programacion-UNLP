  proceso recorrerAvenida( ES contador: numero)
  comenzar
    Pos(PosAv, 1) {PosAv: el robot se mantiene en su avenida actual y se ubica en la calle 1.} 
    repetir 99
      mover
      mientras HayFlorEnLaEsquina
        tomarFlor
        contador := contador + 1
    mientras HayFlorEnLaBolsa
      depositarFlor
  fin
