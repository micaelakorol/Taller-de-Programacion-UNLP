    ok:boolean

    ok := V

    RecibirMensaje(quienSoy,rc)

      BloquearEsquina(11,11) {salto para chequear si hay papel en esa posicion 11,11}
      Pos(11,11)
      si HayPapelEnLaEsquina 
        tomarPapel

        mover

        depositarPapel
        avIni := PosAv
        caIni := PosCa {actualizo las variables a su nueva posicion}
      sino
        ok := F {si no hay papel, corto el bucle, salto a mi esq inicial y libero 11,11}
        Pos(avIni,caIni)
        LiberarEsquina(11,11) 
    EnviarMensaje(quienSoy, rc)
    EnviarMensaje(caIni,rc)

    id,calleFin:numero

    EnviarMensaje(2,r2)
    EnviarMensaje(3,r3)
    EnviarMensaje(4,r4)
    max := -1
    repetir 4
      RecibirMensaje(id,*)
      si id = 1
        RecibirMensaje(calleFin,r1)
      sino
        si id = 2
          RecibirMensaje(calleFin,r2)
        sino
          si id = 3
            RecibirMensaje(calleFin,r3)
          sino 
            RecibirMensaje(calleFin, r4)
      actualizarMax(id,calleFin,max,idMax)
    Informar(idMax)
  fin
