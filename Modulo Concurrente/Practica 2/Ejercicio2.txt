procesos
  proceso escalera(E ancho: numero; E alto:numero; ES cont:numero)
  variables
    flores,papeles:numero
  comenzar
    flores := 0
    papeles := 0
    Pos(PosAv, PosCa)
    repetir 4 {Total de escalones}
      repetir alto {alto de la escalera}
        mover
      derecha
      repetir ancho {ancho}
        mientras HayFlorEnLaEsquina
          tomarFlor
          flores := flores + 1
        mientras HayPapelEnLaEsquina
          tomarPapel
          papeles := papeles + 1
        mover
      repetir 3
        derecha
    si flores > papeles
      cont := cont + 1
  fin
