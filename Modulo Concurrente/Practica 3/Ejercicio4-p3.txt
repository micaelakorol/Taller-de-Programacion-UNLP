robots
  robot robot1  
  variables
    av,ca,quienSoy:numero
    avIni,caIni:numero
    ok:boolean
  comenzar
    avIni := PosAv
    ok := V
    caIni := PosCa
    RecibirMensaje(quienSoy,rc)
    {Si no salto primero a la posicion, el programa nunca empieza}
    {Primero deberia saltar a la posicion}
    mientras ok
      BloquearEsquina(10,10)
      Pos(10,10) {Tengo que ir si o si a la esquina para ver si hay flores}
      si HayFlorEnLaEsquina
        tomarFlor
        BloquearEsquina(11,11)
        Pos(11,11)
        LiberarEsquina(10,10)
        si HayFlorEnLaBolsa
          depositarFlor
        Pos(avIni,caIni)
        LiberarEsquina(11,11)
      sino 
        ok := F
        Pos(avIni,caIni)
        LiberarEsquina(10,10) {si no hay, la libero}
    {enviar que termin√©}
    EnviarMensaje(quienSoy,rc)
  fin
