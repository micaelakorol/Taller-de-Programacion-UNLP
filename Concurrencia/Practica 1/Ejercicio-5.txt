procesos
  proceso recorrerEsquinas(E av:numero; E ca:numero; ES contador:numero)
  comenzar
  {primero poner el mientras y depsues moverse}
    contador := 0
    Pos(av, ca)
    derecha
    repetir 15
      mover
      mientras HayFlorEnLaEsquina
        tomarFlor
        contador := contador + 1
    depositarFlores(contador)
  fin
  
  proceso recorrerAv(E av:numero; E ca:numero; ES contador:numero)
  comenzar 
    contador := 0
    Pos(av,ca)
    repetir 20
      mover
      mientras HayPapelEnLaEsquina
        tomarPapel
        contador := contador + 1
    depositarPapeles(contador)
  fin
  
  proceso depositarPapeles(E contador:numero)
  comenzar
    repetir contador
      depositarPapel
  fin
  
  proceso depositarFlores(E contador:numero)
  comenzar
    repetir contador
      depositarFlor
  fin
